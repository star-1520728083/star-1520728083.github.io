# bind的实现

const obj = {
	name: 1
}

Function.prototype.mybind = function(context){
	// 不是函数调用直接报错
	if (typeof this !== 'function') {
		throw TypeError("Bind must be called on a function");
	}
	const args = [...arguments].slice(1),// 获取传递的参数
				that = this,  // 保存调用者的函数
				nop = function() {}, // 保存原函数的原型
				fn = function() {
					const newargs = Array.prototype.slice.call(arguments) // 返回函数的参数
					return that.apply(this instanceof nop ? this : context, args.concat(newargs))
				}
	nop.prototype = this.prototype; // 指向空函数的原型 => 原函数的原型
	fn.prototype = new nop(); // _protp._protp // 这一步防止修改原函数原型的值

	return fn;
}

function fn(age, sex) {
	console.log(age, sex)
}
