    # 自动构建项目

    const fs = require('fs');
const path = require('path');
var projectData = {
  name: 'app',
  fileData: [
    {
      name: 'js',
      type: 'dir'
    },
    {
      name: 'css',
      type: 'dir'
    },
    {
      name: 'images',
      type: 'dir'
    },
    {
      name: 'index.html',
      type: 'file',
      content: `
      <!DOCTYPE html>
      <html lang="en">
      <head>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width, initial-scale=1.0">
        <meta http-equiv="X-UA-Compatible" content="ie=edge">
        <title>Document</title>
      </head>
      <body>
        hello world
      </body>
      </html>
      `
    }
  ]
}

const rootPath = path.join(__dirname, projectData.name)

// 删除文件夹
function rmdir(dirPath) {
  let files = fs.readdirSync(dirPath) // 读取该文件夹的下面的子文件或者子文件夹
  files.forEach((child) => {
    let childPath = dirPath + '/' + child;
    if (fs.statSync(childPath).isDirectory()) {
      // 判断是否为文件夹
      rmdir(childPath)
    } else {
      // 删除文件
      fs.unlinkSync(childPath)
    }
  })
  fs.rmdirSync(dirPath)
}
// 创建文件夹
function mkdir(rootPath) {
  if (projectData.name) {
    fs.mkdirSync(rootPath);
    const fileData = projectData.fileData;
    if (fileData) {
      fileData.forEach((f) => {
        f.path = rootPath + '/' + f.name;
        switch (f.type) {
          case 'dir':
            fs.mkdirSync(f.path)
            break;
          case 'file':
            fs.writeFileSync(f.path, f.content)
            break;
          default:
            break;
        }
      })
    }
  }
}

fs.stat(rootPath, (err) => {
  if (err) {
    mkdir(rootPath);
  } else {
    rmdir(rootPath);
    mkdir(rootPath);
  }
});
